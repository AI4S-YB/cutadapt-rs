      1012 +    cli: &Cli,                                               
      1013 +    records: Vec<cutadapt_core::record::SequenceRecord>,     
      1014 +    adapters: Vec<cutadapt_core::adapters::Adapter>,         
      1015 +    opener: &FileOpener,                                     
      1016 +    format: FileFormat,

      1018 +    quality_base: i32,                                       
      1019 +    min_len: Option<usize>,

      1021 +) -> Result<(u64, u64, u64, u64, u64, u64, u64, u64)> {

      1065 +    steps.push(Box::new(sink));

           +sum();

           +ers, steps);

           + we use unsafe ptr trick                                     
      1075 +    // for the trimmer. For the sink we need to restructure. 
      1076 +    // Instead, rebuild stats from the pipeline output.      
      1077 +    // Actually the pipeline consumed everything; we need to 
           +write output differently.                                    
      1078 +    // Let's use a different approach: collect output then wr

      1079 +    reads_with_adapter1 = unsafe { (*trimmer_ptr).reads_with_

      1082 +    // collects into a sink. Let's access it via the steps ve
           +c.                                                           
      1083 +    // The pipeline moved the steps, so we can't access them.
      1084 +    // We need to restructure: run pipeline and get sink back

      1085 +    // This requires a redesign. For now, use a simpler direc

      1086 +    drop(pipeline);
